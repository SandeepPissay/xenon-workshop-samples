application/json" -d '{    "querySpec": {        "query": {            "term": {                "matchValue": "com:vmware:xenon:services:common:ExampleService:ExampleServiceState",                "propertyName": "documentKind"            }        }    }}' http://127.0.0.1:8000/core/query-tasksDirectly query the set of example services in the system:curl -X POST -H "Content-Type: application/json" -d '{    "taskInfo": {        "isDirect": true    },    "querySpec": {        "query": {            "term": {                "matchValue": "com:vmware:xenon:services:common:ExampleService:ExampleServiceState",                "propertyName": "documentKind"            }        }    }}' http://127.0.0.1:8000/core/query-tasksQuery the set of example services in the system with document contents:curl -X POST -H "Content-Type: application/json" -d '{    "taskInfo": {        "isDirect": true    },    "querySpec": {        "options": [            "EXPAND_CONTENT"        ],        "query": {            "term": {                "matchValue": "com:vmware:xenon:services:common:ExampleService:ExampleServiceState",                "propertyName": "documentKind"            }        }    }}' http://127.0.0.1:8000/core/query-tasksQuery the set of documents with name "Amanda":curl -X POST -H "Content-Type: application/json" -d '{    "taskInfo": {        "isDirect": true    },    "querySpec": {        "options": [            "EXPAND_CONTENT"        ],        "query": {            "term": {                "matchValue": "Amanda",                "propertyName": "name"            }        }    }}' http://127.0.0.1:8000/core/query-tasksWildcard query for the set of documents whose name contains "man":curl -X POST -H "Content-Type: application/json" -d '{    "taskInfo": {        "isDirect": true    },    "querySpec": {        "options": [            "EXPAND_CONTENT"        ],        "query": {            "term": {                "matchType": "WILDCARD",                "matchValue": "*man*",                "propertyName": "name"            }        }    }}' http://127.0.0.1:8000/core/query-tasksQuery the set of documents containing the key "key1":curl -X POST -H "Content-Type: application/json" -d '{    "taskInfo": {        "isDirect": true    },    "querySpec": {        "options": [            "EXPAND_CONTENT"        ],        "query": {            "term": {                "matchValue": "key1",                "propertyName": "keyValues"            }        }    }}' http://127.0.0.1:8000/core/query-tasksQuery the set of documents containing the value "value1":curl -X POST -H "Content-Type: application/json" -d '{ "value1",                "propertyName": "keyValues"            }        }    }}' http://127.0.0.1:8000/core/query-tasksQuery the set of documents containing the key "key1" and the value    "query": {            "booleanClauses": [                {                    "occurance": "MUST_OCCUR",                    "term": {                        "matchValue": "key1",                        "propertyName": "keyValues"                    }                },                {                    "occurance": "MUST_OCCUR",                    "term": {                        "matchValue": "value1",                        "propertyName": "keyValues"                    }                }            ]        }    }}' http://127.0.0.1:8000/core/query-tasksQuery the
